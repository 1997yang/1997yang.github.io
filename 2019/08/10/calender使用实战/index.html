//获取某日期的日号
public int getDate(Date d){
Calender ca=new Calender();
ca=Calender.getInstance();
ca.setTime(d);
int days=ca.get(Calender.DATE);
return days;
}

//给定日期的前/后n天
//n=-1表示前一天，n=1表示后一天
public Date getDateForDayNum(Date date,int n){
Calender ca=Calender.getInstance();
ca.setTime(date);
ca.add(Calender.DAY_OF_MONTH,n);
Date d=ca.getTime();
return d;
}
